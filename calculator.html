        <h3 id="result"></h3>

    <script>

        function calculate() {

            try {

                let input = document.getElementById("calcInput").value;

                let result = eval(input);

                document.getElementById("result").innerText = "Result: " + result;

            } catch {

                document.getElementById("result").innerText = "Error!";

            }

        }

    </script>
